#Написати функцію, яка знаходить середнє арифметичне значення довільної кількості чисел.
def seredne_arifm(a, *args):
    return(a+sum(args)/(1+len(args)))
print(seredne_arifm(1,2,3))

#Побудувати рекурсивну функцію обчислення чисел Фібоначі до числа введеного користувачем.
def fibonacci(n):
    if n <= 3:
        return n
    elif n <= 0:
        print("Plese enter a positive int")
    else:
        return fibonacci(n-1)+fibonacci(n-2)
print(fibonacci(5))

#Написати програму, яка обчислює дискримінант квадратного рівняння
import math
def diskr(a, b, c):
    d=(b**2)-4*a*c
    if d>0:
        x1=(-b + math.sqrt(d)) / (2 * a)
        x2=(-b - math.sqrt(d)) / (2 * a)
        return(x1,x2)
    elif d==0:
        return(-b / (2 * a))
    else:
        return(0)
print(diskr(a=1, b=-8, c=7))
